// Description: Save behavior detection to history
// Endpoint: POST /api/behavior/save
// Request: { image: string, behavior: string, confidence: number, description: string }
// Response: { detection: BehaviorDetection, message: string }
export const saveBehaviorDetection = async (data: { image: string; behavior: string; confidence: number; description: string }) => {
  // Mocking the response
  return new Promise<{ detection: BehaviorDetection; message: string }>((resolve) => {
    setTimeout(() => {
      resolve({
        detection: {
          _id: Date.now().toString(),
          userId: 'user1',
          image: data.image,
          behavior: data.behavior as unknown,
          confidence: data.confidence,
          description: data.description,
          createdAt: new Date().toISOString(),
        },
        message: 'Detection saved to history!',
      });
    }, 500);
  });
  // Uncomment the below lines to make an actual API call
  // try {

  //   return response.data;

  // }
};
